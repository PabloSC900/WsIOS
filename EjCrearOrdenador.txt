class MemoriaRAM {
  var capAlmacen : Int = 0
  var fabricante : String = ""
  var rgb : Bool = false
}


class Procesador {
  var velocProc : Double = 0
  var numCores : Int = 0
  var numHilos: Int = 0
  var fabricante : String = ""  
}


class UnidadAlmacenam {
  var capAlmacen : Int = 0
  var fabricante : String = "" 
  var tipoUnidad : String = "" 
}


class TarjetaGrafica {
  var ram : Int = 0
  var consumo : Int = 0
  var cudaCores : Int = 0
  var velocidad : Int = 0
}


class Ordenador {
  var precio : Double = 0
  var nombreEquipo : String = "" 
  var modulosRAM : MemoriaRAM = MemoriaRAM()
  var procesador : Procesador = Procesador()

}
